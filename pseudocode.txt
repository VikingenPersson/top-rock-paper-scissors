FUNCTION getComputerChoice
    INIT variable named choiceInt of type int with starting value 0
    INIT variable named choiceStr of type string and starting value ""
    SET choiceInt to a random number between 1 and 3
    IF choiceInt is less than or equal to 1 THEN
        SET choiceStr to "rock"
    ELSEIF choiceInt is less than or equal to 2 THEN
        SET choiceStr to "paper"
    ELSE
        SET choiceStr to "scissors"
    ENDIF
    RETURN choiceStr
ENDFUNCTION

FUNCTION getUserChoice
    INIT variable called choiceStr of type string with starting value ""
    OBTAIN user choice between "rock" "paper" and "scissors"
    SET choiceStr to users choice
    SET choiceStr to lower case
    RETURN choiceStr
ENDFUNCTION

FUNCTION playRound with parameters userChoice and computerChoice
    INIT variable winner of type string with starting value ""
    IF userChoice and computerChoice is the same THEN
        SET winner to "draw"
    ELSEIF userChoice beats computerChoice THEN
        SET winner to "user"
    ELSE
        SET winner to "computer"
    ENDIF
    RETURN winner
ENDFUNCTION

FUNCTION playGame
    INIT two variables userScore and computerScore of type int and starting value of 0
    INIT variable called round of type int and starting value of 0
    WHILE round is less than 5
        INCREMENT round by 1
        INIT two variables userChoice and computerChoice of type str with starting values of ""
        INIT variable called winner of type string and starting value ""
        CALL getComputerChoice and set computerChoice to its return value
        CALL getUserChoice and set userChoice to its return value
        CALL playRound with userChoice and computerChoice as arguments
        SET winner to the return value of playRound
        IF winner is set to "draw" THEN
            PRINT draw round message
        ELSEIF winner is set to "user" THEN
            PRINT user win round message
            INCREMENT userScore by 1
        ELSE
            PRINT computer won round message
            INCREMENT computerScore by 1
        ENDIF
    ENDWHILE
    IF userScore and computerScore is equal THEN
        PRINT draw game message
    ELSEIF userScore is higher than computerScore THEN
        PRINT user game win message
    ELSE
        PRINT computer game win message
    ENDIF
ENDFUNCTION
